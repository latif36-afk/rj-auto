// ============================================================ // RJ AUTO SERVICE - Google Apps Script Backend v2 // ============================================================ const SHEETS = { CLIENTES: 'Clientes', VEHICULOS: 'Vehiculos', ORDENES: 'Ordenes', DETALLEORDENES: 'DetalleOrdenes', CITAS: 'Citas', CAJA: 'Caja', MECANICOS: 'Mecanicos', DIVIDENDOS: 'Dividendos' }; const HEADERS = { Clientes: ['ID','Nombre','Telefono','Email','FechaRegistro'], Vehiculos: ['ID','ClienteID','Placa','Marca','Modelo','Anio','Color','Kilometraje'], Ordenes: ['ID','VehiculoID','ClienteID','MecanicoID','FechaEntrada','FechaSalida','Estado','Descripcion','TotalManoObra','TotalRepuestos','TotalCobrado','Notas'], DetalleOrdenes: ['ID','OrdenID','Tipo','Descripcion','CostoTaller','PrecioCliente','ClienteTrae','Cantidad','SocioGasto','SocioIngreso'], Citas: ['ID','ClienteID','VehiculoID','Fecha','Hora','Motivo','Estado','Notas'], Caja: ['ID','OrdenID','Fecha','Concepto','Categoria','Monto','Tipo','Socio','MetodoPago','Referencia'], Mecanicos: ['ID','Nombre','Telefono','Especialidad','Estado'], Dividendos: ['ID','Periodo','Fecha','Monto','Decision','Notas'] }; // ─── UTILS ─────────────────────────────────────────────────── function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); const headers = HEADERS[name]; if (headers && headers.length > 0) { sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length) .setBackground('#1C1C1C').setFontColor('#F5B800').setFontWeight('bold'); } } return sheet; } function genId(prefix) { return prefix + '-' + new Date().getTime(); } function sheetToObjects(sheet) { const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } function respond(data) { return ContentService.createTextOutput(JSON.stringify({ success: true, data: data })) .setMimeType(ContentService.MimeType.JSON); } function respondErr(msg) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: msg })) .setMimeType(ContentService.MimeType.JSON); } // ─── ROUTER ────────────────────────────────────────────────── function doGet(e) { try { const p = e.parameter; const action = p.action; switch (action) { case 'getAll': return respond(handleGetAll(p.sheet)); case 'getById': return respond(handleGetById(p.sheet, p.id)); case 'getByField': return respond(handleGetByField(p.sheet, p.field, p.value)); case 'getDashboard': return respond(handleDashboard()); case 'getOrdenFull': return respond(handleGetOrdenFull(p.id)); case 'create': return respond(handleCreate(p.sheet, JSON.parse(p.data))); case 'update': return respond(handleUpdate(p.sheet, p.id, JSON.parse(p.updates))); case 'delete': return respond(handleDelete(p.sheet, p.id)); case 'entregarOrden': return respond(handleEntregarOrden(p.id, p.metodo)); default: return respondErr('Accion no reconocida: ' + action); } } catch(e) { return respondErr(e.message); } } // ─── HANDLERS ──────────────────────────────────────────────── function handleGetAll(sheetName) { if (!SHEETS[sheetName.toUpperCase()]) throw new Error('Hoja invalida: ' + sheetName); return sheetToObjects(getSheet(sheetName)); } function handleGetById(sheetName, id) { const rows = sheetToObjects(getSheet(sheetName)); const row = rows.find(r => r.ID === id); if (!row) throw new Error('No encontrado'); return row; } function handleGetByField(sheetName, field, value) { const rows = sheetToObjects(getSheet(sheetName)); return rows.filter(r => String(r[field]).toLowerCase() === String(value).toLowerCase()); } function handleCreate(sheetName, data) { const sheet = getSheet(sheetName); const headers = HEADERS[sheetName]; const prefixes = { Clientes:'CLI', Vehiculos:'VEH', Ordenes:'ORD', DetalleOrdenes:'DET', Citas:'CIT', Caja:'CAJ', Mecanicos:'MEC', Dividendos:'DIV' }; data.ID = genId(prefixes[sheetName] || 'REC'); if (!data.FechaRegistro && sheetName === 'Clientes') { data.FechaRegistro = new Date().toISOString().split('T')[0]; } const row = headers.map(h => data[h] !== undefined ? data[h] : ''); sheet.appendRow(row); return data; } function handleUpdate(sheetName, id, updates) { const sheet = getSheet(sheetName); const headers = HEADERS[sheetName]; const data = sheet.getDataRange().getValues(); const idCol = headers.indexOf('ID'); for (let i = 1; i < data.length; i++) { if (data[i][idCol] === id) { Object.keys(updates).forEach(key => { const col = headers.indexOf(key); if (col >= 0) sheet.getRange(i + 1, col + 1).setValue(updates[key]); }); return { ID: id, updated: true }; } } throw new Error('Registro no encontrado: ' + id); } function handleDelete(sheetName, id) { const sheet = getSheet(sheetName); const headers = HEADERS[sheetName]; const data = sheet.getDataRange().getValues(); const idCol = headers.indexOf('ID'); for (let i = 1; i < data.length; i++) { if (data[i][idCol] === id) { sheet.deleteRow(i + 1); return { deleted: id }; } } throw new Error('Registro no encontrado'); } // ─── ORDEN FULL (orden + detalle) ──────────────────────────── function handleGetOrdenFull(ordenId) { const ordenes = sheetToObjects(getSheet('Ordenes')); const orden = ordenes.find(o => o.ID === ordenId); if (!orden) throw new Error('Orden no encontrada'); const detalles = sheetToObjects(getSheet('DetalleOrdenes')); orden.items = detalles.filter(d => d.OrdenID === ordenId); return orden; } // ─── ENTREGAR ORDEN → genera cobro en caja ─────────────────── function handleEntregarOrden(ordenId, metodo) { const ordenes = sheetToObjects(getSheet('Ordenes')); const orden = ordenes.find(o => o.ID === ordenId); if (!orden) throw new Error('Orden no encontrada'); if (orden.Estado === 'Entregado') throw new Error('Esta orden ya fue entregada'); const detalles = sheetToObjects(getSheet('DetalleOrdenes')); const items = detalles.filter(d => d.OrdenID === ordenId); // Calcular totales const totalManoObra = items .filter(i => i.Tipo === 'ManoObra') .reduce((s,i) => s + (parseFloat(i.PrecioCliente)||0) * (parseFloat(i.Cantidad)||1), 0); const totalRepuestos = items .filter(i => i.Tipo === 'Repuesto' && i.ClienteTrae !== 'SI') .reduce((s,i) => s + (parseFloat(i.PrecioCliente)||0) * (parseFloat(i.Cantidad)||1), 0); const totalCobrado = totalManoObra + totalRepuestos; const hoy = new Date().toISOString().split('T')[0]; // Actualizar orden handleUpdate('Ordenes', ordenId, { Estado: 'Entregado', FechaSalida: hoy, TotalManoObra: totalManoObra, TotalRepuestos: totalRepuestos, TotalCobrado: totalCobrado }); // Registrar cobros en caja (por socio si aplica) if (totalCobrado > 0) { // Cobros generales de mano de obra const moGeneral = itemsMO.filter(i=>!i.SocioIngreso); const moGenTotal = moGeneral.reduce((s,i)=>s+(parseFloat(i.PrecioCliente)||0)*(parseFloat(i.Cantidad)||1),0); if (moGenTotal > 0) { handleCreate('Caja', { OrdenID: ordenId, Fecha: hoy, Concepto: 'M.O. orden ' + ordenId, Categoria: 'Mano de obra', Monto: moGenTotal, Tipo: 'Cobro', MetodoPago: metodo || 'Efectivo', Referencia: ordenId }); } // Cobros de mano de obra por socio const sociosMO = [...new Set(itemsMO.filter(i=>i.SocioIngreso).map(i=>i.SocioIngreso))]; sociosMO.forEach(socio => { const total = itemsMO.filter(i=>i.SocioIngreso===socio).reduce((s,i)=>s+(parseFloat(i.PrecioCliente)||0)*(parseFloat(i.Cantidad)||1),0); if (total > 0) handleCreate('Caja', { OrdenID: ordenId, Fecha: hoy, Concepto: 'M.O. ' + socio + ' - orden ' + ordenId, Categoria: 'Mano de obra', Monto: total, Tipo: 'Cobro', Socio: socio, MetodoPago: metodo || 'Efectivo', Referencia: ordenId }); }); // Cobros de repuestos if (totalRepuestos > 0) { handleCreate('Caja', { OrdenID: ordenId, Fecha: hoy, Concepto: 'Repuestos orden ' + ordenId, Categoria: 'Repuesto vendido', Monto: totalRepuestos, Tipo: 'Cobro', MetodoPago: metodo || 'Efectivo', Referencia: ordenId }); } // Gastos de repuestos por socio (adelantos) const repConGasto = items.filter(i=>i.Tipo==='Repuesto'&&i.ClienteTrae!=='SI'&&i.SocioGasto&&parseFloat(i.CostoTaller)>0); const sociosGasto = [...new Set(repConGasto.map(i=>i.SocioGasto))]; sociosGasto.forEach(socio => { const total = repConGasto.filter(i=>i.SocioGasto===socio).reduce((s,i)=>s+(parseFloat(i.CostoTaller)||0)*(parseFloat(i.Cantidad)||1),0); if (total > 0) handleCreate('Caja', { OrdenID: ordenId, Fecha: hoy, Concepto: 'Gasto repuesto ' + socio + ' - orden ' + ordenId, Categoria: 'Repuesto', Monto: total, Tipo: 'Gasto', Socio: socio, MetodoPago: 'Efectivo', Referencia: ordenId }); }); // Gastos generales de repuestos sin socio const repGeneral = items.filter(i=>i.Tipo==='Repuesto'&&i.ClienteTrae!=='SI'&&!i.SocioGasto&&parseFloat(i.CostoTaller)>0); const totalRepGastoGeneral = repGeneral.reduce((s,i)=>s+(parseFloat(i.CostoTaller)||0)*(parseFloat(i.Cantidad)||1),0); if (totalRepGastoGeneral > 0) handleCreate('Caja', { OrdenID: ordenId, Fecha: hoy, Concepto: 'Costo repuestos orden ' + ordenId, Categoria: 'Repuesto', Monto: totalRepGastoGeneral, Tipo: 'Gasto', MetodoPago: 'Efectivo', Referencia: ordenId }); } return { ordenId, totalCobrado, estado: 'Entregado' }; } // ─── DASHBOARD ─────────────────────────────────────────────── function handleDashboard() { const hoy = new Date().toISOString().split('T')[0]; const ordenes = sheetToObjects(getSheet('Ordenes')); const caja = sheetToObjects(getSheet('Caja')); const citas = sheetToObjects(getSheet('Citas')); const clientes = sheetToObjects(getSheet('Clientes')); const estados = { Pendiente: 0, 'En proceso': 0, Listo: 0, Entregado: 0 }; ordenes.forEach(o => { if (estados[o.Estado] !== undefined) estados[o.Estado]++; }); const ingresosMes = caja .filter(c => c.Tipo === 'Cobro' && c.Fecha && c.Fecha.toString().startsWith(hoy.substring(0,7))) .reduce((sum, c) => sum + (parseFloat(c.Monto) || 0), 0); const citasHoy = citas.filter(c => c.Fecha === hoy).length; return { totalClientes: clientes.length, ordenesActivas: (estados['Pendiente'] || 0) + (estados['En proceso'] || 0), estadosOrdenes: estados, ingresosMes: ingresosMes, citasHoy: citasHoy }; } // ─── SETUP ─────────────────────────────────────────────────── function setupSpreadsheet() { Object.values(SHEETS).forEach(name => getSheet(name)); // Agregar mecánicos por defecto si no existen const mecSheet = getSheet('Mecanicos'); const mecs = sheetToObjects(mecSheet); if (mecs.length === 0) { mecSheet.appendRow(['MEC-001','Ray','','Mecánico','Activo']); mecSheet.appendRow(['MEC-002','Jorge','','Mecánico','Activo']); } SpreadsheetApp.getActiveSpreadsheet().toast('✅ Hojas listas', 'RJ Auto Service', 5); }
